Setup Google Apps Script

Lakukan sekali. Setelah selesai, paste URL Web App ke kolom sync di atas.

1Buat Google Sheet baru. Baris 1 isi header: nama | tipe | lb | harga di kolom A–D. Rename tab sheet jadi Kompetitor.
2Buka Apps Script: menu Extensions → Apps Script. Hapus kode lama, paste kode di bawah ini:
const SHEET_NAME = 'Kompetitor';

function doGet(e) {
  const sh = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME);
  const rows = sh.getDataRange().getValues();
  const headers = rows[0].map(h => String(h).toLowerCase().trim());
  const data = rows.slice(1).map(r => {
    const o = {};
    headers.forEach((h,i) => o[h] = r[i]);
    return o;
  }).filter(r => r.nama);
  return ContentService
    .createTextOutput(JSON.stringify({ok:true, data}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const sh = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME);
  const payload = JSON.parse(e.postData.contents);
  sh.clearContents();
  sh.appendRow(['nama','tipe','lb','harga']);
  payload.data.forEach(r =>
    sh.appendRow([r.nama, r.tipe, r.lb, r.harga])
  );
  return ContentService
    .createTextOutput(JSON.stringify({ok:true, saved:payload.data.length}))
    .setMimeType(ContentService.MimeType.JSON);
}
3Deploy sebagai Web App: Deploy → New deployment → Web App. Set Execute as: Me dan Who has access: Anyone. Klik Deploy → salin URL yang muncul.
4Paste URL ke kolom "Google Sheets" di atas. Klik ↓ Load untuk ambil data dari sheet, atau ↑ Simpan untuk kirim data ke sheet.
